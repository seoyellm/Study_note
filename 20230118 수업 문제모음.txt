```sql
--전체 사원 정보 조회
SELECT * FROM EMP;

--사원번호, 직업, 급여 출력
SELECT EMPNO, JOB, SAL FROM EMP;

--사원번호, 직업, 급여+300 출력
SELECT EMPNO, JOB, SAL, SAL+300 AS UPSAL FROM EMP;

--커미션을 300받는 사원 출력
SELECT * FROM EMP
WHERE COMM=300;

--커미션을 받지 않는 사원 출력
SELECT * FROM EMP
WHERE COMM IS NULL OR COMM=0;

--사원명, 커미션+100 출력
SELECT ENAME, COMM, COMM+100 AS UPCOMM FROM EMP;

--COMM과 급여 합산 출력
SELECT COMM,SAL, COMM+SAL AS SUM FROM EMP;

--SMITH의 직업을 NULL로 변경
UPDATE EMP SET JOB=NULL
WHERE ENAME='SMITH';

--10번 부서 사원 출력
SELECT * FROM EMP
WHERE EMPNO =10;

--JOB이 CLERK인 사원의 사원번호, 사원명, 직업 출력
SELECT EMPNO, ENAME, JOB FROM EMP
WHERE JOB= CLERK;

--SAL 이 3000 이하를 받는 사원 출력
SELECT * FROM EMP
WHERE SAL <=3000;

--SAL 이 3000 이상 5000이하를 받는 사원 출력
SELECT * FROM EMP
WHERE SAL BETWEEN 3000 AND 5000

--B~-C사원명 출력
SELECT ENAME FROM EMP
WHERE ENAME >='B' AND ENAME <D;

--입사일이 1981년도 입사자 출력
SELECT * FROM EMP
WHERE DATE BETWEEN TO_DATE('1981-01-01','YYYY-MM-DD') AND TO_DATE('1981-12-12','YYYY-MM-DD');
WHERE DATE >= '1981-01-01' AND DATE '1981-12-12';
WHERE DATE BETWEEN '1981-01-01' AND '1981-12-12';

--직업이 CLERK 이거나 MANAGER 인 사원 출력
SELECT * FROM EMP
WHERE JOB= 'CLERK' OR JOB='MANAGER';
WHERE JOB IN ('CLERK','MANAGER');

--직업이 CLERK 이거나 MANAGER가 아닌 사원 출력
WHERE JOB NOT IN ('CLERK','MANAGER');

--사원명이 M으로 시작하는 사원 출력
SELECT * FROM EMP
WHERE ENAME LIKE 'M%';

--이름이 M으로 시작하는 사람, SUBSTR 사용
SELECT *FROM EMP
WHERE SUBSTR(ENAME,1,1)='M';

--사원명의 두번째 글자가 A인 사원 출력
WHERE ENAME LIKE '_A%';

--이름의 세번째 글자가 R인경우 출력
WHERE SUBSTR(ENAME,3,1)='R';
WHERE ENAME LIKE '__R%';

--이름의 마지막 글자가 N은 경우 출력
WHERE ENAME LIKE '%N';

--이름에 A가 포함되는 경우
WHERE ENAME LIKE '%A%';

--JOB과 ENAME 잇기
SELECT JOB||ENAME FROM EMP;
SELECT CONCAT(ENAME,JOB) AS EJ FROM EMP;
SELECT CONCAT(CONCAT(ENAME, '-'),JOB) AS EJ FROM EMP;
```
